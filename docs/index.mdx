# Yeoubi UI

React UI Library for Yeoubi.

# Getting Started

## Installation

```console
yarn @yeoubi/ui
```

## Usage

```typescript
import React from 'react';
import { Heading } from '@yeoubi/ui';

const Logo: React.FunctionCompoment<{}> = () => (
  <Heading level={1}>
    Yeoubi
  </Heading>
);
```
